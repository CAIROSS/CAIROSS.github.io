void union(int A[], int B[])
{
    int C[];

    while(a != NULL && b != NULL)
    {
        if (a > b)
            C.addLast(b)
            B.removeFirst()
        else if (a == b)
            C.addLast(b)
            A.removeFirst()
            B.removeFirst()
        else
        {
            C.addLast(a)
            A.removeFirst()
        }
    }
    while(!A.isEmpty())
        a <- A.get(1)
        C.addLast(a)
        A.removeFirst()
    while(!B.isEmpty())
        b <- B.get(1)
        C.addLast(b)
        B.removeFirst()
    return C
}
